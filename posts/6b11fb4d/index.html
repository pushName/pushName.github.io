// HDFS的块信息
@org.junit.Test
public void biks() throws Exception {

    Path i = new Path("/usr/root/helloworld.txt");
    FileStatus iStatus =fs.getFileStatus(i);
    BlockLocation[] biks= fs.getFileBlockLocations(iStatus, 0, iStatus.getLen()); //取块的位置

    for (BlockLocation b : biks) {
        System.out.println(b);

    }

    //上层只能通过流打开i文件
    FSDataInputStream in = fs.open(i);
    for (int j = 0; j < 15; j++) {
        System.out.println((char)in.readByte());
    }

}

//上层计算只能通过流打开i文件
FSDataInputStream in = fs.open(i);

    for (int j = 0; j < 4; j++) {

    }

    in.seek(1048576);//跳到第二个块，第二个块的偏移量是1048576
